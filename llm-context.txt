rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Default deny
    match /{document=**} {
      allow read, write: if false;
    }

    // Helper functions
    function isSignedIn() {
      return request.auth != null;
    }

    function isOwner(userId) {
      return isSignedIn() && request.auth.uid == userId;
    }

    function getUserOrgId() {
      // Returns the orgId of the currently logged in user
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.orgId;
    }

    function isOrgMember(orgId) {
      return isSignedIn() && getUserOrgId() == orgId;
    }

    function incoming() {
      return request.resource.data;
    }

    function existing() {
      return resource.data;
    }

    // Organisations
    match /organisations/{orgId} {
      allow get: if isSignedIn();
      allow create: if isSignedIn() && incoming().ownerId == request.auth.uid;
      allow update: if isOwner(existing().ownerId);
    }

    // Users
    match /users/{userId} {
      // Users can get their own data, or anyone can get anyone in their org
      allow get: if isOwner(userId) || isOrgMember(existing().orgId);
      
      // Users can only list members of their own organisation
      allow list: if isSignedIn() && resource.data.orgId == getUserOrgId(); 
      
      allow create: if isOwner(userId);
      allow update: if isOwner(userId) || (isSignedIn() && exists(/databases/$(database)/documents/organisations/$(existing().orgId)) && get(/databases/$(database)/documents/organisations/$(existing().orgId)).data.ownerId == request.auth.uid);
    }

    // Departments
    match /departments/{deptId} {
      allow read: if isOrgMember(resource.data.orgId);
      allow create, update: if isOrgMember(incoming().orgId);
    }

    // Roles
    match /roles/{roleId} {
      allow read: if isOrgMember(resource.data.orgId);
      allow create, update: if isOrgMember(incoming().orgId);
    }

    // Tasks
    match /tasks/{taskId} {
      allow read: if isOrgMember(resource.data.orgId);
      allow create: if isOrgMember(incoming().orgId) && incoming().assignedBy == request.auth.uid;
      allow update: if isOrgMember(existing().orgId) && (
        (existing().assignedTo == request.auth.uid && incoming().diff(existing()).affectedKeys().hasOnly(['state', 'updatedAt'])) ||
        (existing().assignedBy == request.auth.uid)
      );
    }

    // Chat Messages (NEW)
    match /messages/{messageId} {
      allow read: if isOrgMember(resource.data.orgId);
      allow create: if isOrgMember(incoming().orgId) && incoming().senderId == request.auth.uid;
    }

    // File Vault (NEW)
    match /files/{fileId} {
      allow read: if isOrgMember(resource.data.orgId);
      allow create: if isOrgMember(incoming().orgId) && incoming().uploadedBy == request.auth.uid;
    }

    // Invites
    match /invites/{inviteId} {
      allow get: if true; // Publicly accessible to check token
      allow create: if isSignedIn();
      allow update: if true; // To mark as used
    }

    // Notifications
    match /notifications/{notifId} {
      allow read: if isOwner(resource.data.userId);
      allow create: if isSignedIn();
      allow update: if isOwner(resource.data.userId) && incoming().diff(existing()).affectedKeys().hasOnly(['read']);
    }
  }
}

      const SYSTEM_PROMPT = `
You are the management engine for a recursive organisational system.
You receive an org schema (dept → lead → roles, infinitely nested) and a
management request. You return a JSON decision only — no explanation,
no preamble, no markdown.

Rules:
- A lead has authority over all roles and sub-leads in their dept and below
- A user holding multiple roles has the union of all their permissions
- Help requests route to the direct lead of the dept the task belongs to
- If the requesting user holds roles in multiple depts, resolve by task dept
- Never grant authority upward — a role cannot affect its own lead
- Return { "error": "insufficient_context" } if the schema is incomplete

Always return valid JSON. Never return anything else.
`;

      const response = await ai.models.generateContent({
        model: "gemini-3-flash-preview",
        contents: JSON.stringify({
          request_type,
          org_context,
          payload
        }),
        config: {
          systemInstruction: SYSTEM_PROMPT,
          responseMimeType: "application/json",
          responseSchema: {

              confidence: { type: Type.STRING },
              error: { type: Type.STRING }
            }
          }
        }

import React from 'react';
import { BrowserRouter as Router, Routes, Route, Navigate } from 'react-router-dom';
import { AuthProvider, useAuth } from './contexts/AuthContext';
import Dashboard from './pages/Dashboard';
import Login from './pages/Login';
import RegisterOrg from './pages/RegisterOrg';
import JoinLink from './pages/JoinLink';
import Admin from './pages/Admin';
import { motion, AnimatePresence } from 'motion/react';

function ProtectedRoute({ children }: { children: React.ReactNode }) {
  const { firebaseUser, loading } = useAuth();
  if (loading) return <div className="h-screen w-screen flex items-center justify-center font-mono animate-pulse">ORGFlow</div>;
  if (!firebaseUser) return <Navigate to="/login" />;
  return <>{children}</>;
}

function AdminRoute({ children }: { children: React.ReactNode }) {
  const { isLead, loading } = useAuth();
  if (loading) return null;
  if (!isLead) return <Navigate to="/dashboard" />;
  return <>{children}</>;
}

function AppContent() {
  return (
    <AnimatePresence mode="wait">
      <Routes>
        <Route path="/login" element={<Login />} />
        <Route path="/register/org" element={<RegisterOrg />} />
        <Route path="/join/:token" element={<JoinLink />} />
        <Route
          path="/dashboard/*"
          element={
            <ProtectedRoute>
              <Dashboard />
            </ProtectedRoute>
          }
        />
        <Route
          path="/admin/*"
          element={
            <ProtectedRoute>
              <AdminRoute>
                <Admin />
              </AdminRoute>
            </ProtectedRoute>
          }
        />
        <Route path="/" element={<Navigate to="/dashboard" />} />
      </Routes>
    </AnimatePresence>
  );
}

export default function App() {
  return (
    <AuthProvider>
      <Router>
        <AppContent />
      </Router>
    </AuthProvider>
  );

import React, { createContext, useContext, useEffect, useState } from 'react';
import { onAuthStateChanged, User as FirebaseUser } from 'firebase/auth';
import { doc, getDoc, onSnapshot } from 'firebase/firestore';
import { auth, db } from '../lib/firebase';
import { User, Organisation } from '../types';

interface AuthContextType {
  user: User | null;
  organisation: Organisation | null;
  firebaseUser: FirebaseUser | null;
  loading: boolean;
  isLead: boolean;
}

const AuthContext = createContext<AuthContextType>({
  user: null,
  organisation: null,
  firebaseUser: null,
  loading: true,
  isLead: false,
});

export function AuthProvider({ children }: { children: React.ReactNode }) {
  const [firebaseUser, setFirebaseUser] = useState<FirebaseUser | null>(null);
  const [user, setUser] = useState<User | null>(null);
  const [organisation, setOrganisation] = useState<Organisation | null>(null);
  const [loading, setLoading] = useState(true);
  const [isLead, setIsLead] = useState(false);

  useEffect(() => {
    let leadStatusUnsubscribe: (() => void) | null = null;

    const unsubscribe = onAuthStateChanged(auth, async (fUser) => {
      setFirebaseUser(fUser);
      
      // Clean up previous listener if it exists
      if (leadStatusUnsubscribe) {
        leadStatusUnsubscribe();
        leadStatusUnsubscribe = null;
      }

      if (fUser) {
        // Fetch custom user doc
        const userDoc = await getDoc(doc(db, 'users', fUser.uid));
        if (userDoc.exists()) {
          const userData = { id: userDoc.id, ...userDoc.data() } as User;
          setUser(userData);

          // Fetch org doc
          if (userData.orgId) {
            const orgDoc = await getDoc(doc(db, 'organisations', userData.orgId));
            if (orgDoc.exists()) {
              setOrganisation({ id: orgDoc.id, ...orgDoc.data() } as Organisation);
            }
          }

          // Check if lead
          leadStatusUnsubscribe = onSnapshot(doc(db, 'users', fUser.uid), (doc) => {
             setIsLead(doc.data()?.isLead || false);
          });
          
          setLoading(false);
        } else {
          setUser(null);
          setOrganisation(null);
          setLoading(false);
        }
      } else {
        setUser(null);
        setOrganisation(null);
        setLoading(false);
      }
    });

    return () => {
      unsubscribe();
      if (leadStatusUnsubscribe) {
        leadStatusUnsubscribe();
      }
    };
  }, []);

  return (
    <AuthContext.Provider value={{ user, organisation, firebaseUser, loading, isLead }}>
      {children}
    </AuthContext.Provider>
  );
}

export const useAuth = () => useContext(AuthContext);

import { useAuth } from '../../contexts/AuthContext';

import React, { useEffect, useState, useRef } from 'react';
import { collection, query, where, orderBy, limit, onSnapshot, addDoc, serverTimestamp } from 'firebase/firestore';
import { db } from '../../lib/firebase';
import { useAuth } from '../../contexts/AuthContext';
import { Send } from 'lucide-react';
import { motion } from 'motion/react';
import { cn } from '../../lib/utils';

interface Message {
  id: string;
  text: string;
  senderId: string;
  senderName: string;
  orgId: string;
  createdAt: any;
}

export default function Chat() {
  const { user, organisation } = useAuth();
  const [messages, setMessages] = useState<Message[]>([]);
  const [newMessage, setNewMessage] = useState('');
  const messagesEndRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!organisation) return;

    const q = query(
      collection(db, 'messages'),
      where('orgId', '==', organisation.id),
      orderBy('createdAt', 'asc'),
      limit(100)
    );

    const unsubscribe = onSnapshot(q, (snapshot) => {
      setMessages(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() } as Message)));
      setTimeout(() => messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' }), 100);
    });

    return () => unsubscribe();
  }, [organisation]);

  const sendMessage = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!newMessage.trim() || !user || !organisation) return;

    const text = newMessage.trim();
    setNewMessage('');

    try {
      await addDoc(collection(db, 'messages'), {
        text,
        senderId: user.id,
        senderName: user.name,
        orgId: organisation.id,
        createdAt: serverTimestamp()
      });
    } catch (err) {
      console.error("Error sending message:", err);
    }
  };

  return (
    <div className="flex flex-col h-[calc(100vh-140px)]">
      <div className="mb-4">
        <h1 className="text-2xl font-bold text-neutral-900">Org Chat</h1>
        <p className="text-sm text-neutral-500">Global communication</p>
      </div>

      <div className="flex-1 bg-white border border-neutral-200 rounded-2xl overflow-hidden flex flex-col">
        <div className="flex-1 overflow-y-auto p-4 space-y-4">
          {messages.length === 0 ? (
            <div className="h-full flex items-center justify-center text-neutral-400 text-sm">
              No messages yet. Start the conversation!
            </div>
          ) : (
            messages.map((msg) => {
              const isMe = msg.senderId === user?.id;
              return (
                <motion.div 
                  initial={{ opacity: 0, y: 10 }}
                  animate={{ opacity: 1, y: 0 }}
                  key={msg.id} 
                  className={cn("flex flex-col", isMe ? "items-end" : "items-start")}
                >
                  <span className="text-[10px] text-neutral-400 font-medium mb-1 px-1">
                    {isMe ? 'You' : msg.senderName}
                  </span>
                  <div 
                    className={cn(
                      "px-4 py-2 rounded-2xl max-w-[80%] text-sm",
                      isMe 
                        ? "bg-neutral-900 text-white rounded-tr-sm"

                    )}

                    {msg.text}
                  </div>
                </motion.div>
              );
            })
          )}
          <div ref={messagesEndRef} />
        </div>

        <form onSubmit={sendMessage} className="p-3 bg-neutral-50 border-t border-neutral-100 flex gap-2">

            type="text"
            value={newMessage}
            onChange={(e) => setNewMessage(e.target.value)}

interface FileRecord {
  id: string;
  name: string;
  url: string;
  size: number;
  uploadedBy: string;
  uploadedByName: string;
  orgId: string;
  createdAt: any;
}

export default function FileVault() {
  const { user, organisation } = useAuth();
  const [files, setFiles] = useState<FileRecord[]>([]);
  const [loading, setLoading] = useState(true);
  const [uploading, setUploading] = useState(false);

  const loadFiles = async () => {
    if (!organisation) return;
    setLoading(true);
    try {
      const q = query(
        collection(db, 'files'),
        where('orgId', '==', organisation.id),
        orderBy('createdAt', 'desc')
      );
      const snap = await getDocs(q);
      setFiles(snap.docs.map(d => ({ id: d.id, ...d.data() } as FileRecord)));
    } catch (err) {
      console.error("Error loading files:", err);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    loadFiles();
  }, [organisation]);

  const handleFileUpload = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file || !user || !organisation) return;

    setUploading(true);
    try {
      // Create a unique path in storage
      const fileRef = ref(storage, `orgs/${organisation.id}/${crypto.randomUUID()}_${file.name}`);
      
      // Upload to Firebase Storage
      await uploadBytes(fileRef, file);
      
      // Get the download URL
      const url = await getDownloadURL(fileRef);

      // Save metadata to Firestore
      await addDoc(collection(db, 'files'), {
        name: file.name,
        url,
        size: file.size,
        uploadedBy: user.id,
        uploadedByName: user.name,
        orgId: organisation.id,
        createdAt: serverTimestamp()
      });

      // Refetch files
      await loadFiles();
    } catch (err) {
      console.error("Error uploading file:", err);
      alert("Failed to upload file. Have you enabled Firebase Storage?");
    } finally {
      setUploading(false);
      e.target.value = ''; // Reset input
    }
  };

  const formatSize = (bytes: number) => {
    if (bytes === 0) return '0 B';
    const k = 1024;
    const sizes = ['B', 'KB', 'MB', 'GB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
  };

  return (
    <div className="space-y-6">
      <div className="flex justify-between items-end">
        <div>
          <h1 className="text-2xl font-bold text-neutral-900">File Vault</h1>
          <p className="text-sm text-neutral-500">Shared organisation documents</p>
        </div>
        
        <label className="bg-neutral-900 text-white px-4 py-2 rounded-xl flex items-center cursor-pointer hover:bg-neutral-800 transition-colors">
          {uploading ? <Loader2 className="h-4 w-4 mr-2 animate-spin" /> : <FileUp className="h-4 w-4 mr-2" />}
          {uploading ? 'Uploading...' : 'Upload File'}
          <input type="file" className="hidden" onChange={handleFileUpload} disabled={uploading} />
        </label>
      </div>

      {loading ? (
        <div className="animate-pulse space-y-3">
          {[1,2,3].map(i => <div key={i} className="h-16 bg-neutral-100 rounded-2xl" />)}
        </div>
      ) : files.length === 0 ? (
        <div className="py-20 flex flex-col items-center justify-center text-neutral-300 border-2 border-dashed border-neutral-200 rounded-2xl">

        </div>
      ) : (

import React, { useState } from 'react';
import { useNavigate, Link } from 'react-router-dom';
import { createUserWithEmailAndPassword, updateProfile } from 'firebase/auth';
import { doc, setDoc, serverTimestamp } from 'firebase/firestore';
import { auth, db } from '../lib/firebase';
import { motion } from 'motion/react';
import { Building2, User, Mail, Lock, ArrowRight } from 'lucide-react';

export default function RegisterOrg() {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState('');
  const navigate = useNavigate();

  const handleSubmit = async (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    setLoading(true);
    setError('');

    const formData = new FormData(e.currentTarget);
    const orgName = formData.get('orgName') as string;
    const ownerName = formData.get('ownerName') as string;
    const email = formData.get('email') as string;
    const password = formData.get('password') as string;

    try {
      // 1. Create User
      const userCredential = await createUserWithEmailAndPassword(auth, email, password);
      const user = userCredential.user;

      await updateProfile(user, { displayName: ownerName });

      // 2. Create Organisation
      const orgRef = doc(db, 'organisations', crypto.randomUUID());
      await setDoc(orgRef, {
        name: orgName,
        ownerId: user.uid,
        createdAt: serverTimestamp(),
      });

      // 3. Create Root Department
      const deptRef = doc(db, 'departments', crypto.randomUUID());
      await setDoc(deptRef, {
        name: orgName + ' (Root)',
        orgId: orgRef.id,
        parentId: null,
        leadId: user.uid,
        createdAt: serverTimestamp(),
      });

      // 4. Create User Doc
      await setDoc(doc(db, 'users', user.uid), {
        name: ownerName,
        email: email,
        orgId: orgRef.id,
        roleIds: [], // Root owner has special status
        isLead: true,
        createdAt: serverTimestamp(),
      });

      navigate('/admin');
    } catch (err: any) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="min-h-screen flex items-center justify-center bg-neutral-50 px-4">
      <motion.div 
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        className="max-w-md w-full space-y-8 bg-white p-8 rounded-2xl border border-neutral-200"
      >
        <div className="text-center">
          <h2 className="text-3xl font-bold text-neutral-900 tracking-tight">Create your Organisation</h2>
          <p className="mt-2 text-neutral-500">Start with ORGFlow today</p>
        </div>

        {error && (
          <div className="p-3 bg-red-50 text-red-600 text-sm rounded-lg border border-red-100">
            {error}
          </div>
        )}

        <form className="mt-8 space-y-6" onSubmit={handleSubmit}>
          <div className="space-y-4">
            <div className="relative">
              <Building2 className="absolute left-3 top-3 h-5 w-5 text-neutral-400" />
              <input
                name="orgName"
                type="text"
                required
                className="block w-full pl-10 pr-3 py-2.5 border border-neutral-200 rounded-xl focus:ring-2 focus:ring-neutral-200 focus:border-neutral-400 outline-none transition-all"
                placeholder="Organisation Name"
              />
            </div>
            <div className="relative">
              <User className="absolute left-3 top-3 h-5 w-5 text-neutral-400" />
              <input
                name="ownerName"
                type="text"
                required
                className="block w-full pl-10 pr-3 py-2.5 border border-neutral-200 rounded-xl focus:ring-2 focus:ring-neutral-200 focus:border-neutral-400 outline-none transition-all"
                placeholder="Full Name"
              />
            </div>
            <div className="relative">
              <Mail className="absolute left-3 top-3 h-5 w-5 text-neutral-400" />
              <input
                name="email"
                type="email"
                required
                className="block w-full pl-10 pr-3 py-2.5 border border-neutral-200 rounded-xl focus:ring-2 focus:ring-neutral-200 focus:border-neutral-400 outline-none transition-all"
                placeholder="Email Address"
              />
            </div>
            <div className="relative">
              <Lock className="absolute left-3 top-3 h-5 w-5 text-neutral-400" />
              <input
                name="password"
                type="password"
                required
                className="block w-full pl-10 pr-3 py-2.5 border border-neutral-200 rounded-xl focus:ring-2 focus:ring-neutral-200 focus:border-neutral-400 outline-none transition-all"
                placeholder="Password"
              />
            </div>
          </div>

          <button
            type="submit"
            disabled={loading}
            className="group relative w-full flex justify-center py-3 px-4 border border-transparent text-sm font-semibold rounded-xl text-white bg-neutral-900 hover:bg-neutral-800 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-neutral-900 disabled:opacity-50 disabled:cursor-not-allowed transition-all"
          >
            {loading ? 'Creating...' : 'Register Org'}
            <ArrowRight className="ml-2 h-4 w-4" />
          </button>
        </form>

        <div className="text-center text-sm">
          <Link to="/login" className="text-neutral-500 hover:text-neutral-900">
            Already have an account? Log in

import { Task } from '../types';

export async function resolveManagementDecision(
  requestType: string,
  orgContext: any,
  payload: any
) {
  try {
    const response = await fetch('/api/management/resolve', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        request_type: requestType,
        org_context: orgContext,
        payload: payload
      }),
    });
    return await response.json();
  } catch (error) {
    console.error('AI Management resolution failed:', error);
    return { error: 'failed_to_resolve' };
  }
}
